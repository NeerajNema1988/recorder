const { suite, test } = intern.getPlugin('interface.tdd');
const { assert } = intern.getPlugin('chai');

suite('recorder-generated suite', () => {
  test('Test 1', tst => {
    return tst.remote
      .get('http://example.com')
      .switchToFrame(1)
      .switchToFrame(2)
      .findByXpath('target')
        .moveMouseTo(12, 23)
        .clickMouseButton(0)
        .end()
      .switchToFrame(null)
      .switchToFrame(1)
      .switchToFrame(3)
      .findByXpath('target')
        .moveMouseTo(12, 23)
        .clickMouseButton(0);
  });
});
